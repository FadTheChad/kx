		limit = { 
			OR = {
				has_socialist_government = yes 
				has_government = social_democrat
			}
		}
		OR = {
			owns_state = 1279
			TRK = { TRK_erk_in_power = no }
			TRK = { does_not_exist_or_subject_of_PREV = yes }
		}
	}
	else_if = {
		limit = { has_elected_government = yes }
		OR = {
			owns_state = 1279
			TRK = { TRK_has_jadidist_government = no }
			TRK = { does_not_exist_or_subject_of_PREV = yes }
		}
	}
	else_if = {
		limit = { has_government = national_populist }
		OR = {
			owns_state = 1279
			TRK = { TRK_kurshirmat_in_power = no }
			TRK = { does_not_exist_or_subject_of_PREV = yes }
		}
	}
}
