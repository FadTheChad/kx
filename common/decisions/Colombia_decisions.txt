			}
		}
	
		target_array = global.south_american_tags_array
		
		target_root_trigger = {
			has_country_flag = COL_who_takes_what_participant
		}
		
		target_trigger = {
			can_declare_war_on = FROM
			FROM = {
				is_in_south_america = yes
				has_socialist_government = no
				is_subject = no
				NOT = {	has_country_flag = COL_who_takes_what_participant }
				NOT = { has_country_flag = COL_tag_already_in_sphere }

					is_in_faction = no

				}
			}
		}

		fire_only_once = yes

		ai_will_do = {
			factor = 1000
		}
		ai_hint_pp_cost = who_takes_what_cost_pp_ai?100
		custom_cost_trigger = {
			has_political_power > var:who_takes_what_cost_pp_ai
			command_power > var:who_takes_what_cost_cp_ai
		}
		custom_cost_text = COL_who_takes_what_cost

		cancel_effect = {
			log = "[GetDateText]: [Root.GetName]: Decision cancel COL_who_takes_what target: [From.GetName]"
			ROOT = {
				clr_country_flag = COL_who_takes_what_currently_taking_decision
				remove_from_array = { ROOT.who_take_what_sphere_array = FROM }
				hidden_effect = {
					add_to_variable = { ROOT.who_takes_what_cost_pp = -25 }
					add_to_variable = { ROOT.who_takes_what_cost_pp_ai = -50 }
					add_to_variable = { ROOT.who_takes_what_cost_cp = -10 }
					add_to_variable = { ROOT.who_takes_what_cost_cp_ai = -35 }
					add_political_power = var:who_takes_what_cost_pp
					add_command_power = var:who_takes_what_cost_cp
				}
			}
			FROM = { 
				clr_country_flag = COL_tag_already_in_sphere 
				clr_country_flag = COL_is_taget_of_sphere
			}			
		}
		complete_effect = {
			log = "[GetDateText]: [Root.GetName]: Decision COL_who_takes_what target: [From.GetName]"
			ROOT = {
				set_country_flag = { flag = COL_who_takes_what_currently_taking_decision value = 1 days = 70 }
				add_to_array = { ROOT.who_take_what_sphere_array = FROM }
				hidden_effect = {
					set_temp_variable = { pp_cost = ROOT.who_takes_what_cost_pp }
					multiply_temp_variable = { pp_cost = -1 }
