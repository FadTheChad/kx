WCC_shipping_board_organisation = {
	include = generic_small_fleet_organization
	name = shipping_board
	icon = GFX_idea_shipping_board
	allowed = {
		is_american_tag = yes
	}
	available = {
		FROM = { owns_state = 382 }
	}
	visible = {
		FROM = {
			if = {
				limit = { NOT = { tag = WCC } }
				OR = {
					has_country_flag = USA_naval_companies_allowed
					NOT = { has_global_flag = USA_civil_war_begun_flag }
				}
			}
		}
	}
}
