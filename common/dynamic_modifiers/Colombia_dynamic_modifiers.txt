	icon = "GFX_population_icon"
	enable = { always = yes }
	remove_trigger = { 
		OR = {
				NOT = { is_owned_by = COL }
				any_neighbor_state = {
						owner = { 
							OR = {

								AND = {

								}
							}
						}
				}
				any_neighbor_state = {
						owner = { 
							OR = { 
								NOT = { can_declare_war_on = COL }
