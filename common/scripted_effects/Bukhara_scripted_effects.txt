BUH_change_centralisation = {
	if = {
		limit = { 
			has_country_flag = BUH_centralisation_of_emirate
			BUH_alim_khan_in_power = yes 
		}
		# convert percent to absolute
		if = {
			limit = {
				check_variable = { BUH_centralisation_change > -1 }
				check_variable = { BUH_centralisation_change < 1 }
			}
			multiply_temp_variable = { BUH_centralisation_change = 100 }
		}

		# tooltip
		effect_tooltip = {
			add_to_variable = {
				BUH_centralisation = BUH_centralisation_change
				tooltip = BUH_centralisation_tooltip
			}
		}

		# lower limit = legitimacy * -1
		set_temp_variable = { BUH_lower_limit = BUH_centralisation }
		multiply_temp_variable = { BUH_lower_limit = -1 }

		# upper limit = 100 - legitimacy
		set_temp_variable = { BUH_upper_limit = 100 }
		subtract_from_temp_variable = { BUH_upper_limit = BUH_centralisation }

		# avoid values outside the allowed range
		clamp_temp_variable = {
			var = BUH_centralisation_change
			min = BUH_lower_limit
			max = BUH_upper_limit
		}

		# add the variable
		add_to_variable = { BUH_centralisation = BUH_centralisation_change }
	}
}

BUH_closer_to_chaos = {
	if = {
		limit = {

		if = {
			limit = {
				check_variable = { BUH_chaos > 3 }
			}
			custom_effect_tooltip = BUH_closer_to_chaos_yellow
			add_to_variable = { BUH_chaos = -1 }
		}
		else_if = {
			limit = {
				check_variable = { BUH_chaos < 4 }
				check_variable = { BUH_chaos > 1 }
			}
			custom_effect_tooltip = BUH_closer_to_chaos_red
			add_to_variable = { BUH_chaos = -1 }
		}
		else = {
			custom_effect_tooltip = BUH_closer_to_chaos_dead
			add_to_variable = { BUH_chaos = -1 }
			if = {
				limit = {
					has_country_flag = BUH_young_bukhara_radical
					OR = {
						is_ai = no
						AND = {
							is_ai = yes
							NOT = {
								has_game_rule = {
									rule = BUH_PATH
									option = BUH_PATH_ABS_MON
								}

									rule = BUH_PATH

						}
					}
