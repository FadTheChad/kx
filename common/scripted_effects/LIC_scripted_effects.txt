CEYLON_do_math = {

	set_temp_variable = { CEYLON_math_PRF = PRF.CEYLON_INFLEUNCE }
	set_temp_variable = { CEYLON_math_LIC = LIC.CEYLON_INFLEUNCE }
	set_temp_variable = { CEYLON_math_GEA = GEA.CEYLON_INFLEUNCE }
	
	
	
	subtract_from_temp_variable = { CEYLON_math_LIC = CEYLON_math_PRF }
	set_variable = { CEYLON_math_LIC_SOLUTION_PRF = CEYLON_math_LIC  }
	
	set_temp_variable = { CEYLON_math_LIC = LIC.CEYLON_INFLEUNCE }
	
	subtract_from_temp_variable = { CEYLON_math_LIC = CEYLON_math_GEA }
	set_variable = { CEYLON_math_LIC_SOLUTION_GEA = CEYLON_math_LIC  }
	
	set_temp_variable = { CEYLON_math_LIC = LIC.CEYLON_INFLEUNCE }
	
	
	
	subtract_from_temp_variable = { CEYLON_math_PRF = CEYLON_math_LIC }
	set_variable = { CEYLON_math_PRF_SOLUTION_LIC = CEYLON_math_LIC  }
	
	set_temp_variable = { CEYLON_math_PRF = PRF.CEYLON_INFLEUNCE }
	
	subtract_from_temp_variable = { CEYLON_math_PRF = CEYLON_math_GEA }
	set_variable = { CEYLON_math_PRF_SOLUTION_GEA = CEYLON_math_PRF }
	
	set_temp_variable = { CEYLON_math_PRF = PRF.CEYLON_INFLEUNCE }
	
	
	
	subtract_from_temp_variable = { CEYLON_math_GEA = CEYLON_math_LIC }
	set_variable = { CEYLON_math_GEA_SOLUTION_LIC = CEYLON_math_GEA  }
	
	set_temp_variable = { CEYLON_math_GEA = GEA.CEYLON_INFLEUNCE }
	
	subtract_from_temp_variable = { CEYLON_math_GEA = CEYLON_math_PRF }
	set_variable = { CEYLON_math_GEA_SOLUTION_PRF = CEYLON_math_GEA  }
	
	set_temp_variable = { CEYLON_math_GEA = GEA.CEYLON_INFLEUNCE }
	
	
	
	
	

	
	if = {
		limit = {
			AND = {
				check_variable = {
					var = LIC.CEYLON_INFLEUNCE
					value = 20
					compare = greater_than_or_equals
				}
				check_variable = {
					var = CEYLON_math_LIC_SOLUTION_PRF
					value = 5
					compare = greater_than_or_equals
				}
				check_variable = {
					var = CEYLON_math_LIC_SOLUTION_GEA
					value = 5
					compare = greater_than_or_equals
				}
			}
		}
		set_global_flag = LIC_ceylon_victory
		clr_global_flag = PRF_ceylon_victory
		clr_global_flag = GEA_ceylon_victory
		else = {
			clr_global_flag = LIC_ceylon_victory
		}
	}
	
	if = {
		limit = {
			AND = {
				check_variable = {
					var = PRF.CEYLON_INFLEUNCE
					value = 20
					compare = greater_than_or_equals
				}
				check_variable = {
					var = CEYLON_math_PRF_SOLUTION_LIC
					value = 5
					compare = greater_than_or_equals
				}
				check_variable = {
					var = CEYLON_math_PRF_SOLUTION_GEA
					value = 5
					compare = greater_than_or_equals
				}
			}
		}
		set_global_flag = PRF_ceylon_victory
		clr_global_flag = LIC_ceylon_victory
		clr_global_flag = GEA_ceylon_victory
		else = {
			clr_global_flag = PRF_ceylon_victory
		}
	}
	
	if = {
		limit = {
			AND = {
				check_variable = {
					var = GEA.CEYLON_INFLEUNCE
					value = 20
					compare = greater_than_or_equals
				}
				check_variable = {
					var = CEYLON_math_GEA_SOLUTION_LIC
					value = 5
					compare = greater_than_or_equals
				}
				check_variable = {
					var = CEYLON_math_GEA_SOLUTION_PRF
					value = 5
					compare = greater_than_or_equals
				}
			}
		}
		set_global_flag = GEA_ceylon_victory
		clr_global_flag = LIC_ceylon_victory
		clr_global_flag = PRF_ceylon_victory
		else = {
			clr_global_flag = GEA_ceylon_victory
		}
	}
	
}
