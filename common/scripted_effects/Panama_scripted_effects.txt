PAN_combat_corruption = {
	custom_effect_tooltip = PAN_combat_corruption_tt
	if = {
		limit = { has_idea = PAN_rampant_corruption }
		swap_ideas = {
			remove_idea = PAN_rampant_corruption
			add_idea = PAN_widespread_corruption
		}
	}
	else_if = {
		limit = { has_idea = PAN_widespread_corruption }
		swap_ideas = {
			remove_idea = PAN_widespread_corruption
			add_idea = PAN_modest_corruption
		}
	}
	else_if = {
		limit = { has_idea = PAN_modest_corruption }
		swap_ideas = {
			remove_idea = PAN_modest_corruption
			add_idea = PAN_tolerable_corruption
		}
	}
	else_if = {
		limit = { has_idea = PAN_tolerable_corruption }
		swap_ideas = {
			remove_idea = PAN_tolerable_corruption
			add_idea = PAN_clean_society
		}
	}
}
