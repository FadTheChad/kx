on_actions = {
	on_startup = {
		effect = {
			for_each_scope_loop = {
				array = ALO.owned_states
				add_to_array = { RUS.RUS_kazakhstan_region = THIS }
			}
			for_each_scope_loop = {
				array = UZB.owned_states
				add_to_array = { RUS.RUS_bukhara_region = THIS }
			}
			for_each_scope_loop = {
				array = KHI.owned_states
				add_to_array = { RUS.RUS_khiva_region = THIS }
			}
			for_each_scope_loop = {
				array = TRK.owned_states
				add_to_array = { RUS.RUS_kokand_region = THIS }
			}
			TRK = {
				set_temp_variable = { coalition_partner_var = token:social_democrat }
				add_to_coalition = yes
				set_temp_variable = { coalition_partner_var = token:market_liberal }
				add_to_coalition = yes	
				set_temp_variable = { coalition_partner_var = token:social_conservative }
				add_to_coalition = yes				
				set_temp_variable = { coalition_partner_var = token:authoritarian_democrat }
				add_to_coalition = yes				
			}
		}
	}
	#ROOT = attacking side
	#FROM = defending side
	#fired when two countries end up at war with each other (on_war is fired when a country goes to war against anyone and is not fired again when it enters war against another country unless it went to peace first)
	on_war_relation_added = {
		effect = {
			if = {
				limit = {
					OR = {
						AND = {
							OR = {
								tag = TRK
								tag = UZB
								tag = ALO
								tag = TAJ
								tag = KHI
							}
							FROM = {
								OR = {
									tag = RUS
									tag = TRM
								}
							}
						}
						AND = {
							OR = {
								tag = RUS
								tag = TRM
							}
							FROM = {
								OR = {
									tag = TRK
									tag = UZB
									tag = ALO
									tag = TAJ
									tag = KHI
								}
							}
						}
					}
				}
				if = {
					limit = {
						ROOT = {
							OR = {
								tag = TRK
								tag = UZB
								tag = ALO
								tag = TAJ
								tag = KHI
							}
						}
					}
					ROOT = { activate_mission = TRK_peace_with_russia }
				}
			}
		}
	}
	on_monthly_ALO = {
		effect = {
			if = {
				limit = {
					has_country_flag = ALO_1936_elections
				}
				every_owned_state = {
					ALO_get_polls = yes
				}
			}
		}
	}
}
