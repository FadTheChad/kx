		tag = NGR

	}

		days = 1
	}

	option = {
		name = nigeria.1.a

			factor = 100
		}
		if = {
			limit = {
				owns_state = 878
			}

		}
		if = {
			limit = {
				owns_state = 878
			}
			HAU = { transfer_state = 878 }
		}
		every_owned_state = {
			limit = {
				OR = {
					state = 962
					state = 913
				}
			}
			KBR = { transfer_state = PREV }
		}
		if = {
			limit = {
				country_exists = KBR
				KBR = { NOT = { has_country_flag = MAF_already_exists } }
			}

				add_state_core = 962
				add_state_core = 913
				load_oob = KBR
				add_to_array = { global.MAF_collapse_tags = THIS }
			}
		}
		if = {
			limit = {
				country_exists = HAU
				HAU = { NOT = { has_country_flag = MAF_already_exists } }
			}
			HAU = {
				add_state_core = 878
				load_oob = HAU
			}
		}
		if = {
			limit = {
				country_exists = HAU
				country_exists = NGR
				HAU = { NOT = { has_country_flag = MAF_already_exists } }
			}
			HAU = { declare_war_on = { target = NGR type = annex_everything } } 
		}
		HAU = {
			add_ideas = {
				AFR_post_collapse_instability
				MAF_post_collapse_army
				war_economy
				limited_exports
				limited_conscription
			}
			add_to_array = { global.MAF_collapse_tags = THIS }
		}
		NGR = {
			add_to_array = { global.MAF_collapse_tags = THIS }
		}
	}
}
